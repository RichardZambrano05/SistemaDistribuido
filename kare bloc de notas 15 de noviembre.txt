1. MOSTRAR LA CEDULA, NOMBRE Y APELLIDOS DEL CLIENTE, FECHA, TIPO Y TOTAL DE TRANSACCIONES, 
NOMBRE DEL PRODUCTO, NOMBRE DE LA CATEGORIA, DE AQUELLOS REGISTROS DONDE EL MES DE 
TRANSACCIONES SEA ENERO, EL TOTAL DE TRANSACCION ESTE ENTRE 100 Y 500, EL NOMBRE DEL PRODUCTO INCLUYA UNA A.
SELECT CEDULA_CLI, NOMBRE_CLI, APELLIDO_CLI, MONTHNAME(FECHA_TRA) AS 'MES', TIPO_TRA,TOTAL_TRA, NOMBRE_PRO, NOMBRE_CAT
FROM CLIENTES
INNER JOIN TRANSACCIONES ON CODIGO_CLI=FK_CODIGO_CLI
INNER JOIN DETALLES ON CODIGO_TRA=FK_CODIGO_TRA
INNER JOIN PRODUCTOS ON CODIGO_PRO=FK_CODIGO_PRO
INNER JOIN CATEGORIAS ON CODIGO_CAT=FK_CODIGO_CAT
WHERE  MONTHNAME(FECHA_TRA)='JANUARY' AND (TOTAL_TRA BETWEEN 100 AND 500) AND NOMBRE_PRO LIKE '%A%';


2. MOSTRAR EL NOMBRE  Y APELLIDO DEL CAJERO, TOTAL DE LA TRANSACCION, Y LA FECHA CONCATENADA DE LA SIGUIENTE 
MANERA: 
NOMBRE DEL DIA, NUMERO DE DIA, NOMBRE DEL MES, AÑO DE TRANSACCION.  EJEMPLO DE FECHA EN MAYUSCULAS:
WEDNESDAY,15 DE NOVEMBER DEL 2023. SOLO DE AQUELLOS REGISTROS QUE EL NOMBRE  DEL CAJERO TENGA ENTRE 5 Y 7 
LETRAS Y SU APELLIDO INICIE CON P



SELECT NOMBRE_CAJ, APELLIDO_CAJ, TOTAL_TRA, UPPER(CONCAT(DAYNAME(FECHA_TRA),', ', DAY(FECHA_TRA) ,' DE ',MONTHNAME(FECHA_TRA) ,' DEL ',YEAR(FECHA_TRA),'.')) AS 'NOMBRE DEL DIA, NUMERO DE DIA, NOMBRE DEL MES, AÑO DE TRANSACCION'
FROM CAJEROS
INNER JOIN  TRANSACCIONES ON CODIGO_CAJ=FK_CODIGO_CAJ

WHERE  LENGTH(NOMBRE_CAJ) BETWEEN 5 AND 7 AND APELLIDO_CAJ LIKE 'P%';

3. MOSTRAR LA CEDULA,NOMBRE ,APELLIDO Y FECHA DE INGRESO DEL CAJERO,
NOMBRE DE USUARIO QUE SERA IGUAL A LA CONCATENACION DE LA LETRA 'C', SEGUIDO 
DE SU NUMERO DE CEDULA,ADEMAS MOSTRAR LA CLAVE DE ACCESO QUE SERA IGUAL A LA CONCATENACION 
DE SUS 3 PRIMERAS LETRAS DEL NOMBRE EN MAYUSCULAS, LAS 3 PRIMERAS LETRAS DE SU APELLIDO EN MINUSCULAS
Y EL AÑO DE INGRESO.
EJEMPLO:
1314562348  JUAN  PEREZ  2015/01/23  C1314562348  JUAper2015
 

SELECT CEDULA_CAJ, NOMBRE_CAJ, APELLIDO_CAJ, FECHAINGRESO_CAJ, CONCAT(CEDULA_CAJ),' ',(NOMBRE_CAJ),' ',(APELLIDO_CAJ),' ',(FECHAINGRESO_CAJ),' ',CONCAT(CEDULA_CAJ)
FROM CAJEROS

WHERE  LENGTH(NOMBRE_CAJ) BETWEEN 5 AND 7 AND APELLIDO_CAJ LIKE 'P%';



SELECT CEDULA_CAJ, NOMBRE_CAJ, APELLIDO_CAJ, FECHAINGRESO_CAJ,
       CONCAT('C', CEDULA_CAJ) AS 'NOMBRE_USUARIO',
       CONCAT(UPPER(LEFT(NOMBRE_CAJ, 3)), LOWER(LEFT(APELLIDO_CAJ, 3)), YEAR(FECHAINGRESO_CAJ)) AS 'CLAVE_ACCESO'
FROM CAJEROS;
